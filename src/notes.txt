<script>
document.addEventListener('DOMContentLoaded', function() {
  const headers = document.querySelectorAll('.accordion-header');
  
  headers.forEach(header => {
    header.addEventListener('click', function() {
      const content = this.parentElement.querySelector('.accordion-content');
      const icon = this.querySelector('i');
      const isOpen = content.classList.contains('h-64');
      
      // ZATVOR VŠETKY
      document.querySelectorAll('.accordion-content').forEach(c => {
        c.classList.remove('h-64', 'opacity-100');
        c.classList.add('h-0', 'opacity-0');
        c.style.height = '0px';
      });
      document.querySelectorAll('.fa-chevron-down').forEach(i => {
        i.classList.remove('rotate-180');
      });
      
      // OTVOŔ TÚTO (ak nebola otvorená)
      if (!isOpen) {
        content.classList.remove('h-0', 'opacity-0');
        content.classList.add('h-64', 'opacity-100');
        content.style.height = 'auto';
        icon.classList.add('rotate-180');
      }
    });
  });
});
</script>
